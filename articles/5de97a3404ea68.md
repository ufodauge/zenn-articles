```ts
using lock = await lock("lock-key");
```

と書くためのスニペット

```ts
const lock = async (key: string): Promise<Disposable & { lock: Promise<Lock | null> }> => {
  const { resolve: acquireLock, promise: lockAwaiter } = Promise.withResolvers<Lock | null>();
  const { resolve: releaseLock, promise: releaseLockAwaiter } =
    Promise.withResolvers<void>();
    
  navigator.locks.request(key, async (lock) => {
    acquireLock(lock);
    await releaseLockAwaiter;
  });

  return {
    lock: lockAwaiter,
    [Symbol.dispose]: () => {
      releaseLock();
    },
  };
};
```
