## ↓

```tsx
export const TimerPanel = () => {
  // 全然 Date とかでよい
  // タイマーの操作についてはここでは省略
  const [startAt] = useState(Temporal.Now.instant());
  const timerRef = useRef<HTMLSpanElement>(null);

  useEffect(() => {
    if (timerRef.current === null) {
      return;
    }
    const element = timerRef.current;

    let id: number | undefined = undefined;
    const update = () => {
      element.textContent = startAt
        .until(Temporal.Now.instant())
        .toLocaleTimeString();
      id = requestAnimationFrame(update);
    };

    update();

    return () => {
      if (id) {
        cancelAnimationFrame(id);
      }
    };
  }, [startAt]);

  return <span ref={timerRef} />;
};
```

- 時刻を反映させたい要素のテキストを `requestAnimationFrame` で更新
- あとは各々好きな機能をﾓﾘﾓﾘすれば完成
